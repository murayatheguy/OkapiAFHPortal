### Part 2: Update Fallback Logic
```javascript
// utils/facilityPhotos.js

const OKAPI_PLACEHOLDERS = {
  afh: '/placeholders/okapi-afh-placeholder.svg',
  alf: '/placeholders/okapi-alf-placeholder.svg',
  snf: '/placeholders/okapi-snf-placeholder.svg',
  hospice: '/placeholders/okapi-hospice-placeholder.svg',
  default: '/placeholders/okapi-afh-placeholder.svg' // Default to AFH
};

export function getFacilityPhoto(facility) {
  // Priority 1: Owner uploaded photos
  if (facility.owner_photos?.length > 0) {
    return {
      photo: facility.owner_photos[0],
      source: 'owner'
    };
  }
  
  // Priority 2: Google photos
  if (facility.google_photos?.length > 0) {
    const photo = typeof facility.google_photos[0] === 'string' 
      ? facility.google_photos[0] 
      : facility.google_photos[0].url;
    return {
      photo,
      source: 'google'
    };
  }
  
  // Priority 3: Okapi branded placeholder
  return {
    photo: OKAPI_PLACEHOLDERS[facility.facilityType] || OKAPI_PLACEHOLDERS.default,
    source: 'placeholder'
  };
}
```
